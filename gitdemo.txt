# 列出所有本地分支
$ git branch

# 列出所有本地分支和远程分支
$ git branch -a

# 下载远程仓库的所有变动
$ git fetch 

# 新建一个分支，但依然停留在当前分支
$ git branch [branch-name]

# 新建一个分支，并切换到该分支
$ git checkout -b [branch]
$ git checkout -b feature/Issue#102_baota
Switched to a new branch 'feature/Issue#102_baota'

# 取回远程仓库的变化，并与本地分支合并
$ git pull 

# 合并指定分支到当前分支
$ git merge [branch]

#将远程分支合并到当前分支
git fetch
git rebase origin/master
